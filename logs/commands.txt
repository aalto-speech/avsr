# Find lines containing specific text, and output what is right after them
cat FILE |sed -n -e 's/^.*TEXT //p' > OUTPUT_FILE

# Output the float after the words "Loss total"
sed -n 's/.*Loss total \([0-9]*\.[0-9]*\).*/\1/p'
